3   import java.util.ArrayList;
4   import java.util.Arrays;
5   import java.util.List;
6   import lombok.ToString;

8   @ToString
9   public class Bus {
10  
11      // We expect four ClockedComponents: CoffeeMakerUI, CoffeePot, WarmerPlate, and WaterReservoir
12      private final List<BusComponent<?>> synchedComponents = new ArrayList<>(4);
13  
14      public Bus(BusComponent<?>... components) {
15          synchedComponents.addAll(Arrays.asList(components));
16      }
17  
18      public void update(BusMessage message) {
19          for (BusComponent<?> c : synchedComponents) {
20              c.readBusMessage(message);
21          }
22      }
23  }
